\chapter{Complex Numbers}\label{complex-numbers}

A complex number is an expression of the form \(z = x+iy\), where \(x\) and \(y\) are real numbers. The component \(x\) is called the real part of \(z\), and \(y\) is the imaginary part of \(z\). We will denote these by
\begin{align*}
x &= \operatorname{Re}z, \\
y &= \operatorname{Im}z.
\end{align*}

The set of complex numbers forms the complex plane, which we denote by \(\mathbb{C}\). We denote the set of real numbers by \(\mathbb{R}\), and we think of the real numbers as being a subset of the complex plane, consisting of the complex numbers with imaginary part equal to zero.

The correspondence
\begin{align*}
z = x+iy \longleftrightarrow (x,y)
\end{align*}

is a one-to-one correspondence between complex numbers and points (or vectors) in the Euclidean plane \(\mathbb{R}^2\). The real numbers correspond to the \(x\)-axis in the Euclidean plane. The complex numbers of the form \(iy\) are called purely imaginary numbers. They form the imaginary axis \(i\mathbb{R}\) in the complex plane, which corresponds to the \(y\)-axis in the Euclidean plane.

\subsection{Basic Definitions and Algebra}\label{basic-definitions-and-algebra}

\begin{itemize}
\tightlist
\item
  A complex number has the form \(z = a + ib\) where \(a, b\) are real numbers
\item
  \(a\) is the \textbf{real part}: \(\text{Re}(z) = a\)
\item
  \(b\) is the \textbf{imaginary part}: \(\text{Im}(z) = b\)
\item
  A number is \textbf{purely imaginary} if \(a = 0\)
\item
  A number is \textbf{real} if \(b = 0\)
\item
  Zero is the only number that is both real and purely imaginary
\end{itemize}

\subsection{Arithmetic Operations}\label{arithmetic-operations}

\subsubsection{Addition}\label{addition}

\((a + ib) + (c + id) = (a + c) + i(b + d)\)

\subsubsection{Multiplication}\label{multiplication}

\((a + ib)(c + id) = (ac - bd) + i(ad + bc)\)

\subsubsection{Division}\label{division}

\[\frac{a + ib}{c + id} = \frac{(a + ib)(c - id)}{(c + id)(c - id)} = \frac{ac + bd}{c^2 + d^2} + i\frac{bc - ad}{c^2 + d^2}\]

\subsection{Complex Conjugate}\label{complex-conjugate}

\begin{itemize}
\tightlist
\item
  The conjugate of \(z = a + ib\) is \(\bar{z} = a - ib\)
\item
  Properties:

  \begin{itemize}
  \tightlist
  \item
    \(\overline{z_1 + z_2} = \bar{z}_1 + \bar{z}_2\)
  \item
    \(\overline{z_1 \cdot z_2} = \bar{z}_1 \cdot \bar{z}_2\)
  \item
    \(\overline{\left(\frac{z_1}{z_2}\right)} = \frac{\bar{z}_1}{\bar{z}_2}\)
  \item
    \(z\) is real if and only if \(z = \bar{z}\)
  \item
    \(z\) is purely imaginary if and only if \(z = -\bar{z}\)
  \end{itemize}
\item
  Expressing real and imaginary parts using conjugate:

  \begin{itemize}
  \tightlist
  \item
    \(\text{Re}(z) = \frac{z + \bar{z}}{2}\)
  \item
    \(\text{Im}(z) = \frac{z - \bar{z}}{2i}\)
  \end{itemize}
\end{itemize}

\subsection{Absolute Value (Modulus)}\label{absolute-value-modulus}

\begin{itemize}
\tightlist
\item
  The absolute value or modulus of \(z = a + ib\) is:

  \begin{itemize}
  \tightlist
  \item
    \(|z| = \sqrt{a^2 + b^2} = \sqrt{z\bar{z}}\)
  \end{itemize}
\item
  Properties:

  \begin{itemize}
  \tightlist
  \item
    \(|z| \geq 0\) for all \(z\), with equality if and only if \(z = 0\)
  \item
    \(|z_1 \cdot z_2| = |z_1| \cdot |z_2|\)
  \item
    \(\left|\frac{z_1}{z_2}\right| = \frac{|z_1|}{|z_2|}\) for \(z_2 \neq 0\)
  \item
    \(|z_1 + z_2| \leq |z_1| + |z_2|\) (Triangle inequality)
  \item
    \(\left| |z_1| - |z_2| \right| \leq |z_1 - z_2|\)
  \end{itemize}
\end{itemize}

\subsection{Important Inequalities}\label{important-inequalities}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  \textbf{Triangle Inequality}: \(|z_1 + z_2| \leq |z_1| + |z_2|\)

  \begin{itemize}
  \tightlist
  \item
    Equality holds if and only if \(\frac{z_1}{z_2} > 0\) (i.e., the ratio is positive real)
  \end{itemize}
\item
  \(-|z| \leq \text{Re}(z) \leq |z|\)

  \begin{itemize}
  \tightlist
  \item
    Equality on right when \(z\) is positive real
  \end{itemize}
\item
  \(-|z| \leq \text{Im}(z) \leq |z|\)
\item
  \textbf{Cauchy's Inequality}: \(|a_1b_1 + a_2b_2 + \ldots + a_nb_n|^2 \leq (|a_1|^2 + \ldots + |a_n|^2)(|b_1|^2 + \ldots + |b_n|^2)\)

  \begin{itemize}
  \tightlist
  \item
    Equality holds if and only if the \(a_i\) are proportional to the \(b_i\)
  \end{itemize}
\end{enumerate}

\subsection{Vector Interpretation}\label{vector-interpretation}

\begin{itemize}
\tightlist
\item
  Complex numbers can be represented as vectors from the origin
\item
  Vector addition corresponds to complex addition
\item
  The length of the vector is \(|z|\)
\item
  The distance between points \(z_1\) and \(z_2\) is \(|z_1 - z_2|\)
\end{itemize}

\subsection{Polar Form}\label{polar-form}

\begin{itemize}
\tightlist
\item
  A complex number can be written in \textbf{polar form}:

  \begin{itemize}
  \tightlist
  \item
    \(z = r(\cos\theta + i\sin\theta)\) where \(r = |z|\) and \(\theta = \arg(z)\)
  \item
    \(r\) is the modulus: \(r = |z| = \sqrt{a^2 + b^2}\)
  \item
    \(\theta\) is the \textbf{argument} or \textbf{amplitude}: \(\theta = \arg(z)\)
  \end{itemize}
\item
  Properties of polar form:

  \begin{itemize}
  \tightlist
  \item
    For \(z_1 = r_1(\cos\theta_1 + i\sin\theta_1)\) and \(z_2 = r_2(\cos\theta_2 + i\sin\theta_2)\):
  \item
    \(z_1z_2 = r_1r_2[\cos(\theta_1 + \theta_2) + i\sin(\theta_1 + \theta_2)]\)
  \item
    \(\arg(z_1z_2) = \arg(z_1) + \arg(z_2)\)
  \item
    \(\arg\left(\frac{z_1}{z_2}\right) = \arg(z_1) - \arg(z_2)\)
  \end{itemize}
\end{itemize}

\subsection{De Moivre's Formula}\label{de-moivres-formula}

\begin{itemize}
\tightlist
\item
  For \(z = r(\cos\theta + i\sin\theta)\):

  \begin{itemize}
  \tightlist
  \item
    \(z^n = r^n(\cos n\theta + i\sin n\theta)\)
  \end{itemize}
\item
  For \(r = 1\):

  \begin{itemize}
  \tightlist
  \item
    \((\cos\theta + i\sin\theta)^n = \cos n\theta + i\sin n\theta\)
  \end{itemize}
\end{itemize}

\subsection{Roots of Complex Numbers}\label{roots-of-complex-numbers}

\begin{itemize}
\tightlist
\item
  The \(n\)-th roots of \(z = r(\cos\theta + i\sin\theta)\) are:

  \begin{itemize}
  \tightlist
  \item
    \(\sqrt[n]{z} = \sqrt[n]{r}\left[\cos\left(\frac{\theta + 2\pi k}{n}\right) + i\sin\left(\frac{\theta + 2\pi k}{n}\right)\right]\) for \(k = 0, 1, 2, \ldots, n-1\)
  \end{itemize}
\item
  There are \(n\) distinct \(n\)-th roots of any non-zero complex number
\item
  These roots have equal moduli and are equally spaced around a circle
\item
  For \(z = 1\), the roots are called the \textbf{\(n\)-th roots of unity}
\end{itemize}

\subsection{Equations in the Complex Plane}\label{equations-in-the-complex-plane}

\begin{itemize}
\tightlist
\item
  Circle with center \(a\) and radius \(r\): \(|z - a| = r\)

  \begin{itemize}
  \tightlist
  \item
    In algebraic form: \((z - a)(\bar{z} - \bar{a}) = r^2\)
  \end{itemize}
\item
  Straight line: \(z = a + bt\) where \(t\) is a real parameter

  \begin{itemize}
  \tightlist
  \item
    Two lines are parallel if \(b'\) is a real multiple of \(b\)
  \item
    Lines are orthogonal if \(\frac{b'}{b}\) is purely imaginary
  \item
    The angle between lines is \(\arg\frac{b'}{b}\)
  \end{itemize}
\end{itemize}

\subsection{Stereographic Projection and the Riemann Sphere}\label{stereographic-projection-and-the-riemann-sphere}

The extended complex plane is the complex plane together with the
point at infinity. We denote the it by \(C^* = c\cup\{\infty\}\).
- The extended complex plane includes \(\infty\) and can be represented on a unit sphere
- Under stereographic projection, the point \((x_1, x_2, x_3)\) on the sphere corresponds to:
- \(z = \frac{x_1 + ix_2}{1 - x_3}\)

\begin{itemize}
\tightlist
\item
  The north pole \((0,0,1)\) corresponds to \(\infty\)
\item
  Properties:

  \begin{itemize}
  \tightlist
  \item
    Circles on the sphere correspond to circles or lines in the plane
  \item
    The distance formula: \(d(z,z') = \frac{2|z-z'|}{\sqrt{(1+|z|^2)(1+|z'|^2)}}\)
  \item
    The sphere is often called the \textbf{Riemann sphere}
  \end{itemize}
\end{itemize}

\subsection{Key Formulas and Results}\label{key-formulas-and-results}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  \(z\bar{z} = |z|^2 = a^2 + b^2\)
\item
  \(\frac{1}{z} = \frac{\bar{z}}{|z|^2} = \frac{a - ib}{a^2 + b^2}\)
\item
  \(z^n = r^n(\cos n\theta + i\sin n\theta)\)
\item
  \(\sqrt[n]{z} = \sqrt[n]{r}\left[\cos\left(\frac{\theta + 2\pi k}{n}\right) + i\sin\left(\frac{\theta + 2\pi k}{n}\right)\right]\) where \(k = 0, 1, \ldots, n-1\)
\item
  Triangle inequality: \(|z_1 + z_2| \leq |z_1| + |z_2|\)
\item
  Cauchy's inequality: \(\left|\sum_{i=1}^{n} a_i\bar{b}_i\right|^2 \leq \left(\sum_{i=1}^{n}|a_i|^2\right)\left(\sum_{i=1}^{n}|b_i|^2\right)\)
\item
  The equation of a circle: \(|z - a| = r\)
\item
  The roots of unity: \(\omega^k\) where \(\omega = \cos\frac{2\pi}{n} + i\sin\frac{2\pi}{n}\) and \(k = 0, 1, \ldots, n-1\)
\end{enumerate}

\chapter{Riemann Surfaces.}\label{riemann-surfaces.}

\begin{definition}[Riemann surface]
\protect\hypertarget{def:unnamed-chunk-2}{}\label{def:unnamed-chunk-2}

A \textbf{Riemann surface} is a second‑countable, Hausdorff topological space \(X\) equipped with a \textbf{complex atlas} \(\mathcal{A} = \{(U_\alpha, \varphi_\alpha)\}\) satisfying the following properties:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  Each \(U_\alpha \subseteq X\) is open.
\item
  Each map \(\varphi_\alpha : U_\alpha \to V_\alpha\)
  is a homeomorphism onto an open set \(V_\alpha \subseteq \mathbb{C}\).
\item
  The collection \(\{U_\alpha\}\) covers \(X\).
\item
  For every pair of charts \((U_\alpha, \varphi_\alpha)\) and \((U_\beta, \varphi_\beta)\) with \(U_\alpha \cap U_\beta \neq \varnothing,\)
  the \textbf{transition map}
  \[
  \varphi_\beta \circ \varphi_\alpha^{-1} :
  \varphi_\alpha(U_\alpha \cap U_\beta) \longrightarrow
  \varphi_\beta(U_\alpha \cap U_\beta)
  \]
  is a \textbf{biholomorphism}.
\end{enumerate}

\end{definition}

\begin{remark}
\textbf{Terminology}:
-``holomorphic functions'' are functions which are complex differentiable on an open set in the complex plane.
-A ``biholomorphic map'' is a holomorphic bijection whose inverse is holomorphic.
\end{remark}

\begin{definition}
\protect\hypertarget{def:unnamed-chunk-4}{}\label{def:unnamed-chunk-4}Let \(R_1\) and \(R_2\) be Riemann surfaces. A map \(f : R_1 \to R_2\)
is said to be \textbf{holomorphic} if for every pair of charts \((U, \varphi) \quad \text{on } R_1, \qquad (V, \psi) \quad \text{on } R_2,\) the map \(\psi \circ f \circ \varphi^{-1}\) is holomorphic on its domain (which may be empty).
\end{definition}

Suppose \((U, \varphi)\), \((U', \varphi')\) are charts on \(R_1\), and \((V, \psi)\), \((V', \psi')\) are charts on \(R_2\). On the overlap where all maps are defined, we have:

\begin{itemize}
\tightlist
\item
  the transition maps \(\varphi' \circ \varphi^{-1},\psi' \circ \psi^{-1}\) are \textbf{biholomorphisms} (by condition (5) in the definition of a complex atlas)
\item
  therefore\\
  \[
  \psi' \circ f \circ (\varphi')^{-1}
  = (\psi' \circ \psi^{-1}) 
    \circ (\psi \circ f \circ \varphi^{-1}) 
    \circ (\varphi \circ (\varphi')^{-1})
  \] is holomorphic if and only if \(\psi \circ f \circ \varphi^{-1}\) is holomorphic.
\end{itemize}

Thus the notion of holomorphicity does \textbf{not} depend on the choice of charts.
